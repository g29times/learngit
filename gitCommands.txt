#创建目录
cd learngit
#使用编辑器新建文件
#在Git中添加文件
git add [fileName]
#查看文件状态
git status
#提交改变
git commit -m "[comment]"
#比较差异或文件改动
git diff [fileName]
#查看文件内容
cat [fileName]
#查看提交历史 从最近到最远
git log
#版本 HEAD表示当前版本 上一个版本就是HEAD^ 往上100个版本HEAD~100
#回退上一个版本
git reset --hard HEAD^
#前滚到未来版本
git reset --hard [版本号]
#查看命令历史 命令版本号
git reflog
#查看工作区和版本库里面最新版本的区别
git diff HEAD -- [fileName]
#撤销工作区修改 回到最近一次git commit或git add
git checkout -- [fileName]
#撤销暂存区修改 HEAD表示最新的版本
git reset HEAD [fileName]
#工作区文件删除
rm [fileName]
#git文件删除 需要commit
#git rm [fileName]
#误删的文件恢复
git checkout -- [fileName]
#创建密匙对
ssh-keygen -t rsa -C [emailAdd]
#在github等远程仓库add key之后 准备绑定密匙代理
ssh-agent
#绑定密匙
ssh-add .../id_rsa
#验证密匙绑定情况 出现You've successfully authenticated则成功
ssh -T git@github.com
#把一个已有的本地仓库与远程库关联
git remote add origin git@github.com:g29times/learngit.git
#把本地库的所有内容推送到远程库 -u参数 关联本地和远程master分支
git push -u origin master
#本地作了提交后 推送master分支至远程库
git push origin master
